const cleanText = (text) => {
  if (!text) return '';
  const cleanHTML = text.replace(/<[^>]+>/g, '');
  const cleanSpecialChars = cleanHTML.replace(/[^\w\s]/gi, '');
  return cleanSpecialChars;
};

          subject: firstResult.subject ? firstResult.subject[0] : 'Unknown subject',
        };
